DROP VIEW IF EXISTS VW_BDLY_HOUSEKEEPING_UNIT;
CREATE VIEW VW_BDLY_HOUSEKEEPING_UNIT AS
(SELECT CONCAT(EXPENSE_HOUSEKEEPING_PAYMENT.UNIT_ID,' ','UNIT'),UNIT.UNIT_NO AS UNITNO,EXPENSE_HOUSEKEEPING_PAYMENT.EHKP_PAID_DATE FROM EXPENSE_HOUSEKEEPING_PAYMENT  
JOIN UNIT ON EXPENSE_HOUSEKEEPING_PAYMENT.UNIT_ID = UNIT.UNIT_ID) UNION 
(SELECT CONCAT(EXPENSE_HOUSEKEEPING_PAYMENT.EHKU_ID,' ','HKUNIT') ,
EXPENSE_HOUSEKEEPING_UNIT.EHKU_UNIT_NO,EXPENSE_HOUSEKEEPING_PAYMENT.EHKP_PAID_DATE AS UNITNO FROM EXPENSE_HOUSEKEEPING_PAYMENT  
JOIN EXPENSE_HOUSEKEEPING_UNIT ON EXPENSE_HOUSEKEEPING_PAYMENT.EHKU_ID = EXPENSE_HOUSEKEEPING_UNIT.EHKU_ID)
