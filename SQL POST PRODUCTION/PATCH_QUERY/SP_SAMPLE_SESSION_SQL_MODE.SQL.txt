DROP PROCEDURE IF EXISTS SAMPLE_SESSION_SQL_MODE;
CREATE PROCEDURE `SAMPLE_SESSION_SQL_MODE`(ii INT,dd DECIMAL(5,2),ff varchar(5))
BEGIN
INSERT INTO t4 values(ii,dd,ff);
select * from t4;
END;


DROP PROCEDURE IF EXISTS SAMPLE_SESSION_SQL_MODE_UPDATE;
CREATE PROCEDURE `SAMPLE_SESSION_SQL_MODE_UPDATE`(dd DECIMAL(5,2),ff varchar(5))
BEGIN
UPDATE  t4 SET d=dd,f=ff where i=33;
select * from t4;
END;
