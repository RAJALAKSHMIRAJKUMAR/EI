-- QUERY FOR CHECKING WRONG UNIT_NO ACCESS_CARD


CREATE OR REPLACE VIEW VW_CUSTOMER_LP_DTS AS SELECT CLP.* FROM CUSTOMER_LP_DETAILS CLP,CUSTOMER_ENTRY_DETAILS CED,UNIT_ACCESS_STAMP_DETAILS UASD  WHERE CED.CUSTOMER_ID=CLP.CUSTOMER_ID AND CED.UNIT_ID=UASD.UNIT_ID AND (UASD.UASD_ID=CLP.UASD_ID OR CLP.UASD_ID IS NULL) AND CLP.CED_REC_VER=CED.CED_REC_VER GROUP BY CLP_ID;

SELECT CLP.*
FROM CUSTOMER_LP_DETAILS CLP LEFT JOIN   VW_CUSTOMER_LP_DTS V ON V.CUSTOMER_ID =CLP.CUSTOMER_ID AND V.CED_REC_VER=CLP.CED_REC_VER
WHERE V.CUSTOMER_ID IS NULL;
